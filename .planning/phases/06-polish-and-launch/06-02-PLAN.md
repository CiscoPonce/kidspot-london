<tasks>

<task type="auto">
  <name>Task 1: Configure Backend PM2 Ecosystem</name>
  <files>backend/ecosystem.config.js</files>
  <action>
    Create a PM2 ecosystem configuration for the backend services.
    - Application `kidspot-api`: `script: ./src/server.js`, `instances: 'max'`, `exec_mode: 'cluster'`.
    - Application `kidspot-worker`: `script: ./scripts/run-discovery.js` (or current entry point), `instances: 1`, `exec_mode: 'fork'`.
    - Set `env: { NODE_ENV: 'production' }`.
    - Set `max_restarts: 10` and `restart_delay: 4000` to avoid zombie loops (Pitfall 2 in RESEARCH.md).
  </action>
  <verify>
    <automated>pm2 check backend/ecosystem.config.js</automated>
  </verify>
  <done>Backend ecosystem configuration created.</done>
</task>

<task type="auto">
  <name>Task 2: Configure Frontend PM2 Ecosystem</name>
  <files>frontend/ecosystem.config.js</files>
  <action>
    Create a PM2 ecosystem configuration for the Next.js frontend.
    - Application `kidspot-web`: `script: node_modules/next/dist/bin/next`, `args: 'start'`.
    - Use `instances: 'max'` and `exec_mode: 'cluster'` for production performance.
    - Set `env: { NODE_ENV: 'production', PORT: 3000 }`.
  </action>
  <verify>
    <automated>pm2 check frontend/ecosystem.config.js</automated>
  </verify>
  <done>Frontend ecosystem configuration created.</done>
</task>

<task type="auto">
  <name>Task 3: Update Dockerfiles and Compose for PM2</name>
  <files>backend/Dockerfile, backend/Dockerfile.worker, frontend/Dockerfile, docker-compose.yml</files>
  <action>
    Integrate PM2 into the Docker deployment.
    - In each `Dockerfile`, install `pm2` globally: `RUN npm install -g pm2`.
    - Update `docker-compose.yml` to use `pm2-runtime start ecosystem.config.js` as the command for services.
    - Ensure `NODE_ENV` is passed as `production`.
    - Per D-05 (discretion), ensure logs are directed to stdout for Docker log management.
  </action>
  <verify>
    <automated>grep -r "pm2-runtime" docker-compose.yml && grep "npm install -g pm2" backend/Dockerfile</automated>
  </verify>
  <done>Docker configuration updated to use PM2.</done>
</task>
